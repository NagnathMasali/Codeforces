#include <bits/stdc++.h>
#include <iostream>
using namespace std;
int main() {
  int a,b;
  for(int x,i=0;i<3;i++){
    cin>>x;
    a+=x;
  }
  for(int x,i=0;i<3;i++){
    cin>>x;
    b+=x;
  }
  int n;
  cin>>n;
  int c=(a/5)+(b/10);
  if(a%5!=0)c++;
  if(b%10!=0)c++;
  if(c<=n)
    cout<<"YES";
  else
    cout<<"NO";
}
